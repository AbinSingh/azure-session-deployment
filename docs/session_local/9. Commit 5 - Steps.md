# Step 1 – Update middleware.py

from fastapi import Request

from session_store import get_session

class SessionMiddleware:

    async def __call__(self, request: Request, call_next):

        # Read cookie from the incoming request
        session_id = request.cookies.get("session_id")

        # Default: no authenticated session
        request.state.session = None
        request.state.session_id = None

        if session_id:

            session = get_session(session_id)

            if session:
                request.state.session = session
                request.state.session_id = session_id

        response = await call_next(request)

        return response

# Let's Understand This Line by Line
# Read the cookie
request.cookies.get("session_id")

# Lookup the session

# Attach it to the request

Now every endpoint automatically receives request.state.session without knowing anything about cookies.

This is exactly why middleware exists.

# Step 2 – Register the Middleware

Open app.py.
from middleware import SessionMiddleware

app = FastAPI()

app.middleware("http")(SessionMiddleware())

Let's understand this. FastAPI maintains an internal middleware chain.

When you write app.middleware("http")(SessionMiddleware())

you're saying: Before calling any endpoint, execute this middleware.

# Step 3 – Add /whoami

from fastapi import Request

@app.get("/whoami")
def who_am_i(request: Request):

    return {
        "session_id": request.state.session_id,
        "session": request.state.session
    }

Notice something.

This endpoint
    does not read cookies
    does not access the dictionary
    does not parse headers

It simply trusts the middleware.
